 15 function Server() {
 16     var self = this;
 17 
 18     // These are the functions that perform calls to the server.
 19     // THERE SHOULD NOT BE ANY PROCESSING IN THESE FUNCTIONS!
 20     this.GetDataInfo = function (callback) {
 21         self.DoAsyncGet({c: 'VisualizerDataCenter', m: 'GetDataInfo'}, {}, callback);
 22     };
 23 
 24     this.GetDistinctsForTemporalGroup = function (tGroupType, callback) {
 25         self.DoAsyncGet({c: 'VisualizerDataCenter', m: 'GetTemporalGroupDistincts'},
 26         {'t-group': tGroupType},
 27         callback);
 28     };
 29 
 30     this.GetFilterCount = function (filterParams, callback) {
 31         self.DoAsyncGet({c: 'VisualizerDataCenter', m: 'GetFilterCount'}, filterParams, callback);
 32     };
 33 
 34     this.GetPlayerDataInfo = function (params, callback) {
 35         self.DoAsyncGet({c: 'PlayerDataCenter', m: 'GetPlayerData'}, params, callback);
 36     };
 37 
 38     // performs as many calls as defined by the splitParts and calls callback for each of them
 39     this.LoadData = function (filterParams, splitParts, _callback, _finalCalbback) {
 40         var callback = _callback;
 41         var finalCallback = _finalCalbback;
 42         // for each of the split parts
 43         var callbacksCounter = 0;
 44         for (var i = 0, max = splitParts.length; i < max; i++) {
 45             var thisFilterParam = jQuery.extend({}, filterParams);
 46             var splitPart = splitParts[i];
 47             thisFilterParam['lim-start'] = splitPart.start;
 48             thisFilterParam['lim-take'] = splitPart.take;
 49             self.DoAsyncGet({c: 'VisualizerDataCenter', m: 'GetFilterData'}, thisFilterParam, function (data) {
 50                 // console.log(data, thisFilterParam);
 51                 // count for calling final callback
 52                 callbacksCounter++;
 53                 if (callback)
 54                     //console.log("Callback", callbacksCounter);
 55                     callback(data, thisFilterParam);
 56                 if (callbacksCounter === splitParts.length && finalCallback) {
 57                     finalCallback();
 58                 }
 59             });
 60         }
 61     };
 62 
 63 
 64 
 65     // Rerouter functions
 66     this.DoAsyncGet = function (addressObj, data, callback) {
 67         //console.log('GET: ', addressObj, data);
 68         $.get('http://cse.unl.edu/~surge/index.php/' + addressObj.c + "/" + addressObj.m, data,
 69                 function (raw_result) {
 70                     //console.log('DONE GET: ', addressObj, data);
 71                     if (callback)
 72                         callback(JSON.parse(raw_result));
 73                 });
 74     };
 75 
 76     var MAX_IN_PART = 1250;
 77     this.GetSplitLogic = function (filterCount) {
 78         // gets the split logic based on filter count
 79         // condition: at least 1 part of 100 rows        
 80         if (filterCount <= MAX_IN_PART) { // only one part necessary
 81             return [{start: 0, take: filterCount}];
 82         }
 83 
 84         var partsNeeded = parseInt(filterCount / MAX_IN_PART);
 85         var fitted = (MAX_IN_PART * partsNeeded);
 86         var overflow = filterCount - fitted;
 87         console.log(filterCount, partsNeeded, overflow);
 88         return Enumerable.Range(0, partsNeeded).Select(function (s) {
 89             return {
 90                 start: s * MAX_IN_PART,
 91                 take: MAX_IN_PART
 92             };
 93         }).Concat([{
 94                 start: fitted,
 95                 take: overflow
 96             }]).ToArray();
 97     };
 98

103         if (filterCount <= MAX_IN_PART) { // only one part necessary

105         }

108         var fitted = (MAX_IN_PART * partsNeeded);
109         var overflow = filterCount - fitted;
110         //console.log(filterCount, partsNeeded, overflow);
111         return Enumerable.Range(0, partsNeeded).Select(function (s) {
112             return {
113                 start: s * MAX_IN_PART,
114                 take: MAX_IN_PART
115             };
116         }).Concat([{
117                 start: fitted,
118                 take: overflow
119             }]).ToArray();
120     };
