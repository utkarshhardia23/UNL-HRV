  7 /**
  8  * Global function to start the introduction tour. Also contains all the messages in the intro tour.
  9  * @returns {null}
 10  */
 11 function startIntroTour() {
 12     var intro = introJs();
 13     intro.setOptions({
 14         tooltipPosition: 'auto',
 15         positionPrecedence: ['left', 'right', 'bottom', 'top'],
 16         steps: [
 17             {
 18                 intro: 'Welcome to the Surge Visualizer!'
 19             },
 20             {
 21                 element: $('.gmnoprint:last')[0],
 22                 intro: "Use the map styles to use different types of maps for overlaying the data"
 23             },
 24             {
 25                 element: $('#mnu-overview')[0],
 26                 intro: "This is the overview tools toggle button. Click this to show the overview tools. Click next to see this"
 27             },
 28             {
 29                 element: $('#info-bar')[0],
 30                 intro: "This is the overview tools. Here the unrest categories, the coutries, event count normalization and the layer options can be set. This is also where the area filter option is available."
 31             },
 32             {
 33                 element: $('#mnu-layers')[0],
 34                 intro: "This is the layers toggle button. Click next to see this"
 35             },
 36             {
 37                 element: $('#layers')[0],
 38                 intro: "This is the layers toolbar. The layers available on the data loaded are shown here, based on the Layer by option selected"
 39             },
 40             {
 41                 element: $('#mnu-filters')[0],
 42                 intro: "This will toggle the filters dashboard where the dates and the unrest categories, countries, event count normalization and layer by options can be set."
 43             },
 44             {
 45                 element: $('#mnu-player')[0],
 46                 intro: "This will open the data player in a new tab. The player displays an animated view of the data chronologically"
 47             },
 48             {
 49                 element: $('#mnu-about')   [0],
 50                 intro: "This will open the About page for the Surge project where information about the project is available."
 51             }, 
 52             {
 53                 intro: "That's all! Feel free to play around with the filters and controls!"
 54             }
 55         ]
 56     });
 57     
 58     intro.start();
 59 }
