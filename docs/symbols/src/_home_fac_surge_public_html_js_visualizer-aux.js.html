  7 /**
  8  * Auxiliary object to toggle the dashboards in the Visualizer.
  9  * @type custom
 10  * @author Hariharan Arunachalam
 11  */
 12 var AUX = {
 13     toggleDashboard: function (onComplete, dashboard) {
 14         dashboard.animate({
 15             height: ["toggle", "swing"]
 16         }, {
 17             duration: 600,
 18             specialEasing: {
 19                 width: "linear",
 20                 height: "easeOutBounce"
 21             },
 22             complete: function () {
 23                 if (onComplete)
 24                     onComplete(); // call the oncomplete if it is provided
 25             }
 26         });
 27     }
 28 }
