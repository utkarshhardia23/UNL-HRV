315     this.GetParamters = function () {

324         };
325         // add the range or single point range based on selection type
326         if (params['unrests'].length === 0)
327             return null;
328         if (params['actors'].length === 0)
329             return null;
330         // add the area parameters if given
331         if (self.IsShowingAreaFilter()) {
332             params['area'] = self.AreaCoordinates;
333         }
334         
335 //        if (params['data-sources'].length === 0) 
336 //            return null;
337 
338         try {
339             // reformat the select range and single points to proper dates
340             if (params['t-select'] === 'range') {

355         } catch (e) {

369     this.LoadData = function () {
370         var params = self.GetParamters();
371         if (params === null)
372             return;
373         self.UpdateState(true);
374         self.LoadState(true);
375         // set loaded to 0
376         self.CurrentlyLoadedCount(0);
377         self.LayerManager().Clean();
378         var allPoints = [];
379         var splitLogic = SERVER.GetSplitLogic(self.FilteredCount());
380         SERVER.LoadData(params, splitLogic,
381                 function (data, split) { // piecewise callback
382                     //console.log(split, data);
383                     self.CurrentlyLoadedCount(self.CurrentlyLoadedCount() + data.result.length);
384                     // Hold the data until complete and then split it by layers and pass it to the layer manager
385                     allPoints = Enumerable.From(data.result).
386                             Select(function (s) {
387                                 return {
388                                     location: new google.maps.LatLng(s.lat, s.lng),
389                                     weight: Number(s.data),
390                                     color: s.color === undefined ? 'default' : s.color
391                                 };
392                             }).Concat(allPoints).ToArray();
393                 },
394                 function () { // final callback
395                     // console.log("Final callback", allPoints);
396                     // pass these onto the layer manager
397                     self.LayerManager().Load(allPoints);
398 
399                     // if the dashboard is visible, hide it                    
400                     if (self.IsFilterDashboardVisible())
401                         self.ToggleFilterDashboard();
402 
403                     // unset the loading
404                     self.UpdateState(false);
405                     self.LoadState(false);
406 
407                     // if the area toggle was on, close it
408                     if (self.IsShowingAreaFilter()) {
409                         self.IsShowingAreaFilter(false);
410                     }
411                 });

464 // _obs = the observable to change
465 // _options = the observable array of options
466 // _propName (Optional) = the property name to be used for selections
467 function SelectOptionAutoToggler(_obs, _options, _propName) {
468     var self = this;
469     var obs = _obs;
470     var options = _options;
471     var propName = _propName;
472 
473     this.SelectAllDisable = ko.computed(function () {
474         // if all the options in _obs are already selected, can't select all       
475         return options().length === obs().length;
476     });
477 
478     this.SelectNoneDisable = ko.computed(function () {
479         return obs().length === 0;
480     });
481 
482     this.SelectAll = function () {
483         // clear and add all the items
484         self.SelectNone();
485         if (propName === undefined || propName === null) {
486             obs(Enumerable.From(options()).ToArray()); // no property name
487         } else {
488             obs(Enumerable.From(options()).Select(function (s) {
489                 return s[propName];
490             }).ToArray()); // select the property name
491         }
492 
493     };
494 
495     this.SelectNone = function () {
496         // clear all
497         obs.removeAll();
498     };

567 /**
568  * The view model for managing the layers view model inside the visualizer vm
569  * @param {VisualizerVM} _parent The VisualizerVM parent for back referencing
570  * @param {google.maps.Map} map The reference to the Google Map object.
571  * @returns {LayerManagerVM}
572  */
573 function LayerManagerVM(_parent, map) {
574     var self = this;
575     var parent = _parent;
576     this.Map = map;
577     this.Layers = ko.observableArray();
578 
579     this.IsUsingGlobalControls = ko.observable(false);
580     this.Opacity = ko.observable(10);
581     this.ComputedOpacity = ko.computed(function () {
582         return self.Opacity() / 10;
583     });
584 
585     this.Radius = ko.observable(4);
586 
587     this.MaxMaxIntensity = ko.observable();
588     this.MaxIntensityIndex = ko.observable(0);
589     this.MaxIntensityScale = ko.observableArray();
590     this.MaxIntensity = ko.computed(function () {
591         return self.MaxIntensityScale()[self.MaxIntensityIndex()];
592     });
593     
594     this.Clean = function () {
595         // remove all the layers
596         for (var i = 0, max = self.Layers().length; i < max; i++) {
597             var layer = self.Layers()[i];
598             layer.Destroy();
599         }
600         // remove the layers
601         self.Layers.removeAll();
602         // TODO: Preserve layers if requested
603 
604     };
605 
606     this.Load = function (data) {
607         // split based on the color
608         var splitData = Enumerable.From(data).GroupBy(function (s) {
609             return s.color;
610         }).OrderBy(function(s) {
611             return s.Key();
612         }).ToArray();
613         var lMaxIntensity = 0;
614         for (var i = 0, max = splitData.length; i < max; i++) {
615             var sd = splitData[i];
616             var layer = new LayerVM(self, self.Map, sd.Key(), sd.ToArray(), i);
617             self.Layers.push(layer);
618 
619             if (layer.MaxIntensityValue() > lMaxIntensity)
620                 lMaxIntensity = layer.MaxIntensityValue();
621         }
622         // calculate the global max intensity scale
623         self.MaxIntensityScale(Enumerable.Range(1, Math.ceil(lMaxIntensity)).ToArray());
624         self.MaxMaxIntensity(self.MaxIntensityScale().length - 1);
625         self.MaxIntensityIndex(0);
626     };
627     
628     // subscribe to the globals
629     this.ComputedOpacity.subscribe(function (s) {
630         // change the opacity of all the layers
631         changeLayersProperty('opacity', s);
632     });
633 
634     this.Radius.subscribe(function (s) {
635         changeLayersProperty('radius', s);
636     });
637 
638     this.MaxIntensity.subscribe(function (s) {
639         changeLayersProperty('maxIntensity', s);
640     });
641 
642     function changeLayersProperty(propName, value) {
643         for (var i = 0, max = self.Layers().length; i < max; i++) {
644             var l = self.Layers()[i];
645             l.ChangeHeatMapProperty(propName, value);
646         }
647     };
648 }

650 /**
651  * The view model that manages each of the layers in the {@link LayerManagerVM}
652  * @param {LayerManagerVM} _parent The parent LayerManagerVM for back referencing.
653  * @param {google.maps.Map} map The reference to the Google Map object.
654  * @param {String} name The name of this layer to be displayed in the Layer Manager UI.
655  * @param {Custom} points The data points as sent by the back end to be rendered on this layer.
656  * @param {Number} index The positional index of this layer.
657  * @returns {LayerVM}
658  */
659 function LayerVM(_parent, map, name, points, index) {
660     var self = this;
661     var parent = _parent;
662     this.Map = map;
663     this.HeatMap = null;
664     this.DataCount = ko.observable(points.length);
665     this.Name = ko.observable(name);
666     this.IsActive = ko.observable(true);
667     this.Gradient = ko.observableArray(rgbaGradients[index]);
668     this.IsShowingOptions = ko.observable(false);
669     this.Opacity = ko.observable(10);
670     this.Radius = ko.observable(4);
671     this.ComputedOpacity = ko.computed(function () {
672         return self.Opacity() / 10;
673     });
674     this.MaxIntensityScale = ko.observableArray();
675     this.IsUsingGlobalControls = ko.computed(function () {
676         return parent.IsUsingGlobalControls();
677     });
678     this.ToggleOptions = function () {
679         // close all other options
680         for (var i = 0, max = parent.Layers().length; i < max; i++) {
681             var l = parent.Layers()[i];
682             if (l.Name() !== self.Name())
683                 l.IsShowingOptions(false);
684         }
685         self.IsShowingOptions(!self.IsShowingOptions());
686     };
687 
688 // compute the max, min values
689     var dataEn = Enumerable.From(points);
690     var maxValue = dataEn.Max(function (s) {
691         return s.weight;
692     });
693 
694     this.MaxMaxIntensity = ko.observable();
695     this.MaxIntesityIndex = ko.observable(0);
696     // create a scale with integer values up to max value
697     var x = Enumerable.Range(1, Math.ceil(maxValue)).ToArray();
