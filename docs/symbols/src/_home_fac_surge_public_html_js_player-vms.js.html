  9 /**
 10  * The main view model for the Surge player view.
 11  * @class
 12  * @param {google.maps.Map} map Reference to the Google Map object.
 13  * @returns {PlayerVM}
 14  * @author Hariharan Arunachalam
 15  */
 16 function PlayerVM(map) {
 17     
 18     /**
 19      * Variable to be used to reference properties and methods on the {@link PlayerVM} when inside
 20      * methods where closure would eclipse usage of {@link this}
 21      * @private @readonly
 22      * @type PlayerVM
 23      */
 24     var self = this;
 25     
 26     /**
 27      * A reference to the Google map object for manipulating the map display.
 28      * @private @readonly
 29      * @type {google.maps.Map}
 30      */
 31     this.Map = map;
 32     
 33     /**
 34      * The reference to the google.maps.visualization.HeatmapLayer object.
 35      */
 36     this.HeatMap = null; // the heatmap layer
 37 
 38     /**
 39      * The selected Temportal Group for filtering the data for this animation.
 40      */
 41     this.SelectedTemporalGroup = ko.observable('day');
 42 
 43     /**
 44      * The available Unrest Categories for filtering the data for this animation.
 45      */
 46     this.UnrestCategories = ko.observableArray();
 47     
 48     /**
 49      * The available Countries for filtering the data.
 50      */
 51     this.Countries = ko.observableArray();
 52     
 53     /**
 54      * The selected Countries for filtering the data for this animation.
 55      */
 56     this.SelectedCountries = ko.observableArray();
 57     
 58     /**
 59      * The selected Unrest Categories for filtering the data for this animation.
 60      */
 61     this.SelectedUnrestCategories = ko.observableArray();
 62 
 63     /**
 64      * The array of items that change based on the {@link SelectedTemporalGroup} fpr filtering the data for this animation.
 65      */
 66     this.RangeItems = ko.observableArray();
 67 
 68     /**
 69      * The selected start of the range index with reference to {@link RangeItems}.
 70      */
 71     this.SelectedRangeStartIndex = ko.observable(0);
 72     
 73     /**
 74      * The selected end of the range index with reference to {@link RangeItems}.
 75      */
 76     this.SelectedRangeEndIndex = ko.observable(0);
 77 
 78     /**
 79      * Computed function to convert the range start index to a JavaScript Number. This is used for binding.
 80      */
 81     this.SelectedRangeStartNumber = ko.computed(function () {
 82         return Number(self.SelectedRangeStartIndex());
 83     });
 84 
 85     /**
 86      * The Selected Temporal group based formatting option.
 87      */
 88     this.SelectedTemporalBasedFormat = ko.computed(function () {
 89         switch (self.SelectedTemporalGroup()) {
 90             case 'day':
 91                 return 'D-MM-YYYY';
 92             case 'month':
 93                 return 'MMMM, YYYY';
 94             case 'year':
 95                 return 'YYYY';
 96         }
 97     });
 98 
 99     /**
100      * The Selected Temporal group based output format for formatting the date sent to the back end.
101      */
102     this.SelectedTemporalBasedOutputFormat = ko.computed(function () {
103         switch (self.SelectedTemporalGroup()) {
104             case 'day':
105                 return 'MMMM Do, YYYY';
106             case 'month':
107                 return 'MMMM, YYYY';
108             case 'year':
109                 return 'YYYY';
110         }
111     });
112 
113     /**
114      * Function to convert the item at {@link index} in the {@link RangeItems} to the corresponding display format based on the {@link SelectedTemporalGroup}.
115      * @param {Number} index
116      * @returns {String}
117      */
118     this.FormatTemporalItemBasedOnIndex = function (index) {
119         var condition = (self.RangeItems() === undefined || self.RangeItems()[index] === undefined);
120         var r = condition ?
121                 '-' :
122                 moment(self.RangeItems()[index].item, self.SelectedTemporalBasedFormat()).format(self.SelectedTemporalBasedOutputFormat());
123         //console.log(index, condition, self.SelectedTemporalBasedFormat(), condition? 'x' : self.RangeItems()[index].item, r);
124         return r;
125     };
126 
127     /**
128      * Computed function to convert the {@link SelectedRangeStartIndex} to the corresponding formatted display name using the {@link FormatTemporalItemBasedOnIndex}.
129      */
130     this.SelectedRangeStart = ko.computed(function () {
131         return self.FormatTemporalItemBasedOnIndex(self.SelectedRangeStartIndex());
132     });
133     
134     /**
135      * Computed function to convert the {@link SelectedRangeStartIndex} to the corresponding formatted display name using the {@link FormatTemporalItemBasedOnIndex}.
136      */
137     this.SelectedRangeEnd = ko.computed(function () {
138         return self.FormatTemporalItemBasedOnIndex(self.SelectedRangeEndIndex());
139     });
140 
141     this.PlayMode = ko.observable(0); // 0: Input mode, 1: Preparing mode, 2: Playing mode
142 
143     this.FrameDuration = ko.observable("1000"); // time between frame changes
144     this.FrameStepDuration = ko.observable(100); // 100 ms
145     this.FrameSteps = ko.computed(function () {
146         return Number(self.FrameDuration()) / self.FrameStepDuration();
147     });
148 
149     this.Frames = ko.observableArray();
150     this.PrepareMessage = ko.observable("Preparing...");
151 
152     this.LoadedFrames = ko.computed(function () {
153         return Enumerable.From(self.Frames()).Where(function (s) {
154             return s.IsReady();
155         }).Count();
156     });
157 
158     this.AllFramesLoaded = ko.computed(function () {
159         return self.LoadedFrames() === self.Frames().length;
160     });
161 
162     this.Init = function () {
163         // get the categories, countries 
164         SERVER.GetDataInfo(function (data) {
165             self.UnrestCategories(data.UnrestCategories);
166             self.Countries(data.Actors);
167 
168             // select all the unrest categories
169             self.SelectedUnrestCategories(Enumerable.From(self.UnrestCategories()).ToArray());
170 
171             // Select all actors
172             self.SelectedCountries(Enumerable.From(self.Countries()).Select(function (s) {
173                 return s.code;
174             }).ToArray());
175 
176             // initialize stuff here
177             applySubscriptions();
178             self.SelectedTemporalGroup('month');
179 
180         });
181     };
182 
183     function applySubscriptions() {
184         self.SelectedTemporalGroup.subscribe(function (s) {
185             console.log(s);
186             SERVER.GetDistinctsForTemporalGroup(self.SelectedTemporalGroup(), function (result) {
187                 console.log(result);
188                 self.RangeItems(result);
189                 self.SelectedRangeStartIndex(0);
190                 self.SelectedRangeEndIndex(self.RangeItems().length - 1);
191 
192             });
193         });
194 
195         // subscribe to the change in frame
196         self.CurrentFrame.subscribe(function (s) {
197             // when the frame is changed, set the current frame to degrade and the next frame as fresh
198             self.Frames()[self.CurrentFrame()].DisplayFrameData(self.HeatMap, false);
199             if (self.CurrentFrame() + 1 < self.Frames().length)
200                 self.Frames()[self.CurrentFrame() + 1].DisplayFrameData(self.HeatMap, true);
201 
202         });
203     }
204 
205     this.IsPlaying = ko.observable(false);
206 
207     this.Play = function () {
208         self.IsPlaying(true);
209         self.BeginPlay();
210     };
211 
212     this.Pause = function () {
213         if (self.IntervalHandle !== undefined && self.IntervalHandle !== null) {
214             clearTimeout(self.IntervalHandle);
215         }
216         self.IsPlaying(false);
217     };
218 
219     this.Stop = function () {
220         self.Pause();
221         self.CurrentFrame(0);
222         self.PlayStep(0);
223     };
224 
225     this.CurrentFrameRaw = ko.observable(0);
226     
227     this.CurrentFrame = ko.computed(function(){
228         return Number(self.CurrentFrameRaw());
229     });
230     this.PlayStep = ko.observable(0);
231 
232     this.LoadPlayer = function () {
233         self.PlayMode(2);
234         if (self.HeatMap === undefined || self.HeatMap == null) { // initialize the heat map
235             self.HeatMap = new google.maps.visualization.HeatmapLayer({
236                 map: self.Map,
237                 opacity: 1,
238                 radius: 15,
239                 maxIntensity: 100
240             });
241             console.log("Heatmap loaded");
242         }
243         self.CurrentFrameRaw(0);
244         self.PlayStep(0);
245         // load the first frame as staling
246 //        self.Frames()[self.CurrentFrame()].DisplayFrameData(self.HeatMap, false);
247 //        self.Frames()[self.CurrentFrame() + 1].DisplayFrameData(self.HeatMap, true);
248 //        // begin playing
249 //        self.BeginPlay();
250     };
251 
252     this.LoadedFramesPercent = ko.computed(function () {
253         return (self.LoadedFrames() / self.Frames().length * 100) + "%";
254     });
255 
256     this.IntervalHandle = null;
257     this.BeginPlay = function () {
258         // create and start the time out
259         if (self.IntervalHandle !== undefined && self.IntervalHandle !== null) {
260             clearTimeout(self.IntervalHandle);
261         }
262         self.NextStepLoader();
263     };
264     
265     this.SetTimeout = function () {
266         self.IntervalHandle = setTimeout(self.NextStepLoader, self.FrameStepDuration());
267     };
268 
269     this.NextStepLoader = function () {
270         if (self.PlayStep() === self.FrameSteps()) { // requires next frame
271             self.PlayStep(0); // reset the play step
272             if (self.CurrentFrame() + 1 === self.Frames().length) {
273                 // terminate
274                 return;
275             }
276             self.CurrentFrameRaw(self.CurrentFrame() + 1); // increment the frame
277 
278         } else {
279             self.FrameTransitionLoader();
280         }
281         self.PlayStep(self.PlayStep() + 1); // increment the play step
282         self.SetTimeout();
283     };
284 
285     this.FrameTransitionLoader = function () {
286         var allPoints = Enumerable.From([]);
287         var useMax = 1;
288         // set the currently fresh frame to increase
289         if (self.CurrentFrame() + 1 < self.Frames().length) {
290             var fFrame = self.Frames()[self.CurrentFrame() + 1];
291             fFrame.Data = Enumerable.From(fFrame.Data).Select(function (s) {
292                 s.weight += s.incdec;
293                 return s;
294             }).ToArray();
295             //useMax = useMax > fFrame.MaxOnFrame() ? useMax : fFrame.MaxOnFrame();
296             allPoints = Enumerable.From(allPoints).Concat(fFrame.Data);
297         }
298 
299         // set the currently stale frame to decrease
300         var cFrame = self.Frames()[self.CurrentFrame()];
301         cFrame.Data = Enumerable.From(cFrame.Data).Select(function (s) {
302             if (s.weight <= s.incdec)
303                 s.weight = 0;
304             else
305                 s.weight -= s.incdec;
306             return s;
307         });
308         //useMax = useMax > cFrame.MaxOnFrame() ? useMax : cFrame.MaxOnFrame();
309         allPoints = allPoints.Concat(cFrame.Data).ToArray();
310         self.HeatMap.setData(allPoints);
311         //self.HeatMap.set('maxIntensity', useMax);
312     };
313 
314     this.CurrentFrameName = ko.computed(function () {
315         if (self.Frames() === null || self.Frames().length === 0)
316             return "No frames..";
317         return self.Frames()[self.CurrentFrame()].Name();
318     });
319 
320     this.Prepare = function () {
321         self.PlayMode(1);
322         // request the values from the server
323         self.PrepareMessage("Retreiving frame information..");
324         var params = {
325             't-group': self.SelectedTemporalGroup(),
326             'countries': self.SelectedCountries(),
327             'unrests': self.SelectedUnrestCategories()
328         };
329 
330         params['t-start'] = moment(self.RangeItems()[self.SelectedRangeStartIndex()].item, self.SelectedTemporalBasedFormat()).format(settings.mysqlFormat);
331         params['t-end'] = moment(self.RangeItems()[self.SelectedRangeEndIndex()].item, self.SelectedTemporalBasedFormat()).add(1, temporalGroupEdgeAdjustments[self.SelectedTemporalGroup()].add).subtract(1, 'seconds').format(settings.mysqlFormat);
332 
333 
334         SERVER.GetPlayerDataInfo(params, function (d) {
335             console.log(d);
336             var format = temporalGroupFormat[self.SelectedTemporalGroup()];
337             console.log(format);
338             // load the frames
339             self.LoadPlayer(); // load the player and wait for buffering
340             for (var i = 0, max = d.data.length; i < max; i++) {
341                 var r = d.data[i];
342                 // create a new frame
343                 var f = new FrameVM(self);
344                 f.RequiredDataCount(Number(r.count));
345                 f.Number(i);
346                 f.TemporalGroup = self.SelectedTemporalGroup();
347                 f.Date = r.date;
348                 f.FrameTransitionLength = self.FrameSteps();
349                 var m = moment(r.date, settings.mysqlFormat);
350                 f.StartDate = m.clone().format(settings.emptyMysqlFormat);
351                 f.EndDate = moment(f.StartDate, settings.emptyMysqlFormat).clone().add(1, temporalGroupEdgeAdjustments[self.SelectedTemporalGroup()].add).subtract(1, 'seconds').format(settings.mysqlFormat);
352                 // set the name based on the date and selected grouping
353                 f.Name(m.clone().format(format));
354                 if (i === 0)
355                     console.log(f);
356                 self.Frames.push(f);
357                 f.LoadFrameData(function () {
358                     // check if all frames are loaded
359                     if (self.AllFramesLoaded()) {
360 //                        self.LoadPlayer();
361                     }
362                 });
363             }
364 
365 
366         });
367     };
368 }

370 /**
371  * 
372  * @param {PlayerVM} _parent The PlayerVM parent for this frame. Used for back referencing.
373  * @returns {FrameVM}
374  */
375 function FrameVM(_parent) {
376     var self = this;
377     var parent = _parent;
378 
379     this.FrameTransitionLength = 5;
380 
381     this.Data = new Array();
382 
383     this.RequiredDataCount = ko.observable(0);
384     this.LoadedDataCount = ko.observable(0);
385 
386     this.Number = ko.observable(-1);
387     this.Name = ko.observable("");
388     this.Date = null;
389     this.StartDate = null;
390     this.EndDate = null;
391     this.TemporalGroup = "";
392 
393     this.IsReady = ko.computed(function () {
394         return self.Number() !== -1 && self.RequiredDataCount() === self.LoadedDataCount();
395     });
396 
397     this.MaxOnFrame = ko.observable(1);
398 
399     this.LoadFrameData = function (callback) {
400         // create the params object
401         var params = {
402             't-group': self.TemporalGroup,
403             't-select': 'range',
404             't-filter-start': self.StartDate,
405             't-filter-end': self.EndDate,
406             'unrests': parent.SelectedUnrestCategories(),
407             'actors': parent.SelectedCountries(),
408             'ec-norm': 'N'
409         };
410         var splitLogic = SERVER.GetSplitLogic(self.RequiredDataCount());
411         var allPoints = [];
412         SERVER.LoadData(params, splitLogic,
413                 function (data, split) {
414                     allPoints = Enumerable.From(data.result).Select(function (s) {
415                         var w = Number(s.data);
416                         if (w > self.MaxOnFrame())
417                             self.MaxOnFrame(w);
418                         return {
419                             location: new google.maps.LatLng(s.lat, s.lng),
420                             fresh: true,
421                             target: w,
422                             incdec: w / self.FrameTransitionLength,
423                             weight: 0
424                         };
425                     }).Concat(allPoints).ToArray();
426                     self.LoadedDataCount(self.LoadedDataCount() + data.result.length);
427                 }, function () {
428             self.Data = allPoints;
429             if (callback)
430                 callback();
431         });
432     };
433 
434     this.DisplayFrameData = function (heatmap, isFresh) {
435         console.log("Frame loading: ", self.Number());
436         // directly put the data into the heatmap and set all to not fresh
437         self.Data = Enumerable.From(self.Data).Select(function (s) {
438             s.weight = isFresh ? 0 : s.target;
439             return s;
440         }).ToArray();
441         heatmap.setData(self.Data);
442         console.log("Frame loaded: ", self.Number());
443     };
